#  4/8   Controlar la transición de peralte en tramos sin clotoide             #
#  5/8   Analisis de la transición de peraltes entre curvas en S               #
#  6/8   Transición de peralte en curvas en C (evita bombeo en rectas cortas)  #
#  7/8   No acompañar la transicion al bombeo (Autopistas y Autovias)          #

####### ---------------------------------------------------------------------- #
# 1/8 #  Análisis de la mínima longitud de transición al peralte               #
####### ---------------------------------------------------------------------- #
# La pendiente relativa de borde max. limita el peralte que se puede alcanzar  #
# si la clotoide es corta.  peralte = long * ipmax / ancho                     #
# -----   ---     ------------------------------------------------------------ #
<GRT1-2>
# ipmax   0.6     # ipmax = 1.8 - 0.01 * Vp (0.6 para veloc de proyecto 120)   #
</GRT1-2>
<GRT3>
# ipmax   1.0     # ipmax = 1.8 - 0.01 * Vp (1.0 para veloc de proyecto 80)    #
</GRT3>
####### ---------------------------------------------------------------------- #
# 2/8 #  Limite máximo de tramos con peralte menor que el bombeo               #
####### ---------------------------------------------------------------------- #
# Maxima longitud de la transicion de bombeo si rasante < mras.                #
# Especifica la transición al bombeo de curvas-rectas / clotoide-recta         #
# También, en el caso de clotoides grandes se inserta un dato de peralte igual #
# a bombeo evitando tramos en los que pueda acumularse agua                    #
#                                                                              #
# Si el tramo con peralte menor que el bombeo es mas largo que Mlbo se inserta #
# un dato de peralte con valor bombeo a Lbo metros.                            #
#                                                                              #
# Solo se analiza si la pendiente de la rasante es inferior a mras             #
# Si mras = 0.0 controla siempre sin importar rasante.                         #
#                                                                              #
#       Mlbo  mras  Lbo                                                        #
#------ ----  ----  ---- ----------------------------------------------------- #
<GRT1-2>
 Klmbom 40.0  0.00  40.0                                                       #
</GTR1-2>
<GRT3>
 Klmbom 20.0  0.00  20.0                                                       #
</GRT3>

# 5/8 #  Analisis de la transición de peraltes entre curvas en S               #
####### ---------------------------------------------------------------------- #
# Para Rectas de longitud menor a la que se define, se realiza la transición   #
# de peralte de +bombeo a -bombeo en una longitud dada y centrada en la recta. #
# La transicion del resto del peralte se realiza linealmente hasta la curva    #
# haya o no clotoides, eliminándose ésta si resulta concava.                   #
# Tambien se aplica en caso de no existir rectas ( curvas de radio grande )    #
#                                                                              #
# Cambio de Peralte de Bombeo curvas en S  Long recta. Long cambio.            #
# ---------------------------------------  ----------- ----------------------- #
<GRT1-2>
  D_CPBS                                   200.        80.                     #
</GRT1-2>
<GRT3>
  D_CPBS                                   150.        40.                     #
</GRT3>

# 6/8 #  Transición de peralte en curvas en C (evita bombeo en rectas cortas)  #
####### ---------------------------------------------------------------------- #
# Para rectas de longitud inferior a uno dado, en lugar de bombeo se deja un   #
# tramo con peralte = valor bombeo en curva y el sentido de giro de las curvas #
#                                                                              #
# La transicion a este peralte se realiza automáticamente desde el punto de    #
# la clotoide que tenga el valor del radio que se corresponde al 'bombeo'      #
#                                                                              #
# Si falta alguna de las dos clotoides (o las dos), se prueba con una          #
# transicion de longitud Rmtr; si la recta que queda tiene longitud menor      #
# que el valor de control, no se realiza la transición al bombeo quedando el   #
# peralte de valor el bombeo en el punto que corresponda                       #
#                                                                              #
# Evitar Bombeo en curvas en C (Ovoides)   Long Max Recta.                     #
#----------------------------------------  ----------------------------------- #
