# ------  ----   ------------------------------------------------------------- #
# En el calculo de la pendiente relativa de borde se puede utilizar el ancho   #
# de calzada que está en el menú de anchos de la calzada, o utilizar los       #

  AC_gen : Activa el uso de AC y NC de la solapa general para el cálculo       #

# planta para cada categoria de carretera, o en una tabla TA para tener en     #
# cuenta numero de carriles. Para activar este calculo, se usa el              #

# Correccion de peraltes si la planta tiene clotoides cortas o no las tiene    #
# Corrige diseños fuera de la Norma; pero si no hay espacio ...                #
#                                                                              #
# Si hay clotoide; pero es corta para el ipmax, se intenta alargar la          #
# transicion de peralte hacia la curva, comprobando que en el contacto         #
# clotoide-circulo el peralte no se rebaja más de dPipmax:                     #
#  0.- Calcula la pendiente de borde y si es mayor que la ipmax, se realiza el #
#      estudio de la pendiente de borde con un peralte reducido en dPipmax a   #
#      la entrada de la curva                                                  #
#  1.1.- si la pendiente de borde aun es mayor que ipmax, se rebaja el peralte #
#        hasta el valor de ipmax. La transicion se deja en la clotoide y anota #
#        un informe de la reduccion del peralte.                               #
#  1.2.- si la pendiente de borde es mayor que ipmax, se estudia si entra el   #
#        peralte total con la rampa ipmax, utilizando hasta 1/dLipmax de la    #
#        circular. Si entra, toma lo que se necesita de la circular.           #
#  2.1.- si aún tomando 1/dLipmax de la circular se supera el ipmax, y         #
#        dNipmax==0 entonces se reduce el peralte y da un informe              #
#  2.2.- si dNipmax==1 se alarga la transicion de peralte hasta 1/dLipmax con  #
#        la ip que entre, aunque se supere la ipmax pero sin rebajar el peralte#
#  2.3.- si dNipmax==2 se alarga la transicion hasta 1/dLipmax de la circular  #
#        y busca cumplir la ipmax tomando el resto necesario en la recta si    #
#        cabe.                                                                 #
#  2.4.- si no cabe, utiliza 1/dLipmax de la circular y la parte de recta      #
#        disponible. La pendiente de borde es mayor que ipmax; pero no baja el #
#        peralte                                                               #
#  3.- Si la recta es corta y se solapan las transiciones del bombeo se lleva  #
#      el inicio de la transicion del peralte a la mitad de la recta y se      #
#      recupera parte de la longitud de transicion que invadia la circular     #
#  4. -  Oct 2013(d) Cuando las curvas son cortas y las clotoides tambien, no  #
#        se puede conseguir la ipmax (pendiente de borde) y la longitud Lmcur  #
#        de peralte constante en la curva a la vez. Con dNipmax a 2 se rebaja  #
#        el peralte para conseguir ipmax y Lmcur. Con dNipmax a 3 no se rebaja #
#        el peralte, se rebaja Lmcur para buscar la ipmax y en ultimo extremo  #
#        se rebasa tambien el ipmax.                                           #
#                                                                              #
  dPipmax  4.0      # Valor a restar al peralte nominal de curva por ipmax     #
  dLipmax  3.0      # Puede entrar hasta 1/3 de la longitud del circulo        #

# dNipmax  1        # 1:Superar la ipmax en casos extremos.NO rebajar peralte  #
# dNipmax  2        # 2:Dejar ipmax y  peralte y tomar el resto en la recta    #

#######  Para los casos en los que no hay clotoide se especifica una longitud  #
#        de transición de peralte, definiendose la parte en recta y en curva   #
#        La longitud que se define es utilizada por el programa para realizar  #
#        la transicion desde el peralte de la curva hasta el valor que define  #
#        trbo.                                                                 #
#        Puede calcularse este valor de 3 formas diferentes  :                 #
#                                                                              #
# Si la transicion cabe en la recta ponerla toda en la recta; si no cabe       #
# utilizar Rmtr y Cmtr                                                         #
# RmtrSC                                                                       #
# A.- En función del valor de máxima pendiente relativa de borde               #
# (IPmax especificada anteriormente según una tabla de valores)                #
  RmtrIP  1.0 : Rmtr=K*Pe*NC*AC*ACfac/ipmax   Frac. en recta o curva mayor Pe  #
  CmtrIP  0.0 : Cmtr=K*Pe*NC*AC*ACfac/ipmax   Frac. en curva o curva menor Pe  #
# ----- --------  ------------------------------------------------------------ #
# B.- Usando una velocidad de variación del peralte :                          #
#                                                                              #
# RmtrK   0.04   : Rmtr = 0.04 * Pe * V     Fraccion en recta o curva mayor Pe #
# CmtrK   0.04   : Cmtr = 0.04 * Pe * V     Fraccion en curva o curva menor Pe #
# ----- --------  ------------------------------------------------------------ #
# C.- Usando un valor fijo calculado según formula                             #
#     long =   L0r  + L1r * per_cur                                            #
# ------   ----  ----                                                          #
#  Rmtr   15.0   0.0           Fraccion en recta (o curva de menor peralte)    #
#  Cmtr    0.0   0.0           Fraccion en curva (o curva de mayor peralte)    #
# ------   ----  ----                                                          #
#                                                                              #
# Pe = dif de peralte CURVA - peralte cero en valor absoluto                   #
# AC = Ancho de calzada nominal (sin sobreanchos)                              #
# Independientemente de la formula usada, el valor calculado siempre queda     #
